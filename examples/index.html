Returns success or failure:

type CheckoutResult =
  | { status: 'success'; payload: BoltTransactionSuccess }
  | { status: 'closed' }

// Usage:
const result = await Charge.checkout(url)
if (result.status === 'success') {
  // Handle successful transaction
  console.log('Reference:', result.payload.reference)
} else if (result.status === 'closed') {
  // Handle user closure
  console.log('User closed the checkout')
}
