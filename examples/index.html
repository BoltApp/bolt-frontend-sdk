Initialize the SDK and handle checkout events:

// Initialize the SDK
BoltSDK.initialize({
  gameId: 'com.knights-of-valor.game',
  publishableKey: 'your-publishable-key',
  environment: 'Development'
})

// Listen to checkout events
BoltSDK.on('checkout-link-succeeded', ({ session }) => {
  console.log('Transaction Completed:', session)
})

BoltSDK.on('checkout-link-failed', ({ session }) => {
  console.log('Transaction failed:', session)
})

BoltSDK.on('checkout-link-closed', ({ session }) => {
  console.log('Checkout closed:', session)
})

  target: 'newTab'
